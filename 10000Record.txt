SELECT TOP 10000
    IDENTITY(int,1,1) id,
    0 nm
INTO #tran
FROM sys.all_columns ac1
CROSS JOIN sys.all_columns ac2;

INSERT INTO department (Name, Description)
SELECT  
    'Name - ' + CONVERT(VARCHAR(10), t.id) 
        + ' of ' + CONVERT(VARCHAR(10), COUNT(t.id) OVER (PARTITION BY t.nm)) AS Name,
    'Description - ' + CONVERT(VARCHAR(10), t.id) 
        + ' of ' + CONVERT(VARCHAR(10), COUNT(t.id) OVER (PARTITION BY t.nm)) AS Description
FROM #tran t;

DROP TABLE #tran;
