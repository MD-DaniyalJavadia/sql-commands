-- Cursor, Variables, aur Loop ko 200 Master Entries ke liye set karein
DECLARE @i INT = 1;
DECLARE @MCodeValue INT = 1;
DECLARE @CurrentDate DATETIME = GETDATE();
DECLARE @NewMasterID INT;
-- @VoucherPrefix iski ab zaroorat nahi hai, kyunki Voucher Computed Column hai.
-- DECLARE @VoucherPrefix NVARCHAR(50) = 'OP-' + FORMAT(@CurrentDate, 'yyyyMMdd') + '-'; 
DECLARE @ActCode1 INT, @ActCode2 INT;

-- ActCode IDs ko ek Array mein Load karein
DECLARE @ActCodeList TABLE (ID INT IDENTITY(1,1), ActCode INT);
INSERT INTO @ActCodeList (ActCode) SELECT ActCodeID FROM #ActCodes WHERE ActCodeID <> 0;
DECLARE @MaxActCodeIndex INT = (SELECT COUNT(*) FROM @ActCodeList);

WHILE @i <= 200
BEGIN
    -- 1. AccountOpeningM mein Entry
    -- !!! Voucher column ko hata diya gaya hai !!!
    INSERT INTO AccountOpeningM (VDate, Remarks, BookCode, VoucherNo, MCode, CompanyID, CurrentDate, InputType, UserID)
    VALUES (
        @CurrentDate, 
        'System Generated Entry ' + CAST(@i AS NVARCHAR(10)), 
        NULL, 
        NULL, 
        @MCodeValue,  -- MCode 1 se shuru hokar +1 hota jayega
        1,             -- CompanyID 1
        @CurrentDate, 
        'OP', 
        NULL 
    );

    -- Master Entry ki ID prapt karein
    SET @NewMasterID = SCOPE_IDENTITY();

    -- ActCodes Sequentially/Randomly chunein
    SELECT @ActCode1 = ActCode FROM @ActCodeList WHERE ID = ((@i - 1) * 2 % @MaxActCodeIndex) + 1;
    SELECT @ActCode2 = ActCode FROM @ActCodeList WHERE ID = ((@i - 1) * 2 + 1) % @MaxActCodeIndex + 1;

    -- 2. AccountOpeningD mein pehli Entry (Debit)
    INSERT INTO AccountOpeningD (PersonID, Debit, Credit, ChkDate, ActCode, Remarks, Cheque, CreateDate, UserID)
    VALUES (
        @NewMasterID,  -- PersonID Master Entry ki ID hogi
        (RAND() * 10000) + 100, -- Random Debit Value
        NULL,          -- Credit NULL
        NULL, 
        @ActCode1,     -- ActCode JSON Data se
        'Debit Entry for MCode ' + CAST(@MCodeValue AS NVARCHAR(10)), 
        NULL, 
        @CurrentDate, 
        NULL
    );

    -- 3. AccountOpeningD mein doosri Entry (Credit)
    INSERT INTO AccountOpeningD (PersonID, Debit, Credit, ChkDate, ActCode, Remarks, Cheque, CreateDate, UserID)
    VALUES (
        @NewMasterID,  -- PersonID Master Entry ki ID hogi
        NULL,          -- Debit NULL
        (RAND() * 10000) + 100, -- Random Credit Value
        NULL, 
        @ActCode2,     -- ActCode JSON Data se
        'Credit Entry for MCode ' + CAST(@MCodeValue AS NVARCHAR(10)), 
        NULL, 
        @CurrentDate, 
        NULL
    );

    -- Counters ko Update karein
    SET @i = @i + 1;
    SET @MCodeValue = @MCodeValue + 1;
END

-- Temporary table ko hata dein
DROP TABLE #ActCodes;

-- Verification Queries (Optional)
SELECT 'Total AccountOpeningM Entries: ' + CAST(COUNT(*) AS NVARCHAR(10)) FROM AccountOpeningM WHERE MCode >= 1 AND MCode < @MCodeValue;
SELECT 'Total AccountOpeningD Entries: ' + CAST(COUNT(*) AS NVARCHAR(10)) FROM AccountOpeningD WHERE PersonID IN (SELECT Id FROM AccountOpeningM WHERE MCode >= 1 AND MCode < @MCodeValue);
